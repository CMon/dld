  dx = x2-x1
  dy = y2-y1

  d = sqrt( sqr(dx) + sqr(dy) )

  a = ( sqr(r1) - sqr(r2) + sqr(d) )/( 2*d )

  h2 = sqr(r1) - sqr(a)
   h = sqrt(h2);
